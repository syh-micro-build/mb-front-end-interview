安装和使用 ​

  • 获取代码
bash
git clone https://github.com/syh-micro-build/mb-front-end-interview.git
  • 安装依赖
bash
cd mb-front-end-interview

npm run boot
  • 运行
bash
npm run docs:dev
  • 打包
bash
npm run docs:build
  • 打包后预览
bash
npm run docs:preview
  • 提交代码
bash
git add .

npx cz
  • 格式化 md 文件
bash
# 检查所有文件
npm run lint

# 格式化所有文件
npm run fix

# 格式化某个文件
npx markdownlint-cli2 src/questions/front-base/3_JavaScript.md --fix
